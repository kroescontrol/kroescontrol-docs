## 🔐 Toegang tot Beschermde Secties

Voor toegang tot interne documentatie:

import { useSession, signIn } from 'next-auth/react'

export function LoginStatus() {
  const { data: session, status } = useSession()
  
  if (status === 'loading') {
    return <div style={{padding: '16px', background: '#f3f4f6', borderRadius: '8px', margin: '16px 0'}}>
      <p>🔄 Sessie wordt geladen...</p>
    </div>
  }
  
  if (session) {
    return <div style={{padding: '16px', background: '#dcfce7', borderRadius: '8px', margin: '16px 0'}}>
      <p>✅ <strong>Ingelogd als:</strong> {session.user?.name || session.user?.email}</p>
      <p>Je hebt toegang tot alle beschermde secties!</p>
    </div>
  }
  
  return <div style={{padding: '16px', background: '#fef3c7', borderRadius: '8px', margin: '16px 0'}}>
    <p>🔑 <strong>Niet ingelogd</strong></p>
    <p>Voor toegang tot interne documentatie:</p>
    <button 
      onClick={() => signIn()} 
      style={{
        padding: '8px 16px', 
        background: '#1e40af', 
        color: 'white', 
        border: 'none', 
        borderRadius: '6px', 
        cursor: 'pointer',
        marginTop: '8px'
      }}
    >
      Inloggen met GitHub/Google
    </button>
  </div>
}

<LoginStatus />
