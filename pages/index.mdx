export function WelcomeMessage() {
  const { data: session } = useSession()
  
  if (session) {
    return (
      <div style={{
        padding: '20px',
        background: 'linear-gradient(135deg, #dbeafe 0%, #e0e7ff 100%)',
        borderRadius: '12px',
        margin: '32px 0',
        textAlign: 'center'
      }}>
        <h3 style={{ margin: '0 0 8px 0' }}>Welkom terug, {session.user?.name || 'Kroescontroller'}! 🎉</h3>
        <div style={{ margin: 0, opacity: 0.8 }}>Je hebt volledige toegang tot alle documentatie secties.</div>
      </div>
    )
  }
  
  return (
    <div style={{
      padding: '20px',
      background: 'linear-gradient(135deg, #fef3c7 0%, #fed7aa 100%)',
      borderRadius: '12px',
      margin: '32px 0',
      textAlign: 'center'
    }}>
      <h3 style={{ margin: '0 0 8px 0' }}>Welkom bij Kroescontrol Docs! 🌟</h3>
      <div style={{ margin: 0, opacity: 0.8 }}>
        Verken onze publieke documentatie of log in voor toegang tot interne secties.
      </div>
    </div>
  )
}
